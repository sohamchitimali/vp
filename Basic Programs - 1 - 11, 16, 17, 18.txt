1.Write a  C# program to check if a number is prime.

using System;

class Program
{
    static void Main()
    {
        Console.Write("Enter a number: ");
        int num = int.Parse(Console.ReadLine());
        bool isPrime = true;

        if (num <= 1)
            isPrime = false;
        else
        {
            for (int i = 2; i <= Math.Sqrt(num); i++)
            {
                if (num % i == 0)
                {
                    isPrime = false;
                    break;
                }
            }
        }

        Console.WriteLine(isPrime ? $"{num} is a Prime number." : $"{num} is not a Prime number.");
    }
}


2.Write a C# program to print the Fibonacci series up to n terms.

using System;

class Program
{
    static void Main()
    {
        Console.Write("Enter the number of terms: ");
        int n = int.Parse(Console.ReadLine());
        int a = 0, b = 1, c;

        Console.WriteLine("Fibonacci Series:");
        for (int i = 1; i <= n; i++)
        {
            Console.Write(a + " ");
            c = a + b;
            a = b;
            b = c;
        }
    }
}


3.Write a  C# program to check whether a string is a palindrome.

using System;

class Program
{
    static void Main()
    {
        Console.Write("Enter a string: ");
        string str = Console.ReadLine();
        string rev = "";

        for (int i = str.Length - 1; i >= 0; i--)
            rev += str[i];

        if (str.Equals(rev, StringComparison.OrdinalIgnoreCase))
            Console.WriteLine("The string is a palindrome.");
        else
            Console.WriteLine("The string is not a palindrome.");
    }
}


4.Write a C# program to reverse a number.

using System;

class Program
{
    static void Main()
    {
        Console.Write("Enter a number: ");
        int num = int.Parse(Console.ReadLine());
        int rev = 0;

        while (num > 0)
        {
            int digit = num % 10;
            rev = rev * 10 + digit;
            num /= 10;
        }

        Console.WriteLine("Reversed Number: " + rev);
    }
}

8.Write a C# program to split a substring from a given string.

using System;

class Program
{
    static void Main()
    {
        Console.Write("Enter a string: ");
        string input = Console.ReadLine();

        Console.Write("Enter the start index: ");
        int start = int.Parse(Console.ReadLine());

        Console.Write("Enter the length: ");
        int length = int.Parse(Console.ReadLine());

        if (start >= 0 && start + length <= input.Length)
        {
            string sub = input.Substring(start, length);
            Console.WriteLine("Extracted substring: " + sub);
        }
        else
        {
            Console.WriteLine("Invalid range for substring!");
        }

        Console.WriteLine("\nSplit example:");
        string[] parts = input.Split(' ');
        foreach (string part in parts)
        {
            Console.WriteLine(part);
        }
    }
}

10.C# Write a C# program to find the maximum occurring character in a given string.

using System;
using System.Collections.Generic;

class Program
{
    static void Main()
    {
        Console.Write("Enter a string: ");
        string input = Console.ReadLine().ToLower();

        Dictionary<char, int> freq = new Dictionary<char, int>();
        foreach (char c in input)
        {
            if (char.IsLetterOrDigit(c))
            {
                if (freq.ContainsKey(c))
                    freq[c]++;
                else
                    freq[c] = 1;
            }
        }

        char maxChar = '\0';
        int maxCount = 0;
        foreach (var pair in freq)
        {
            if (pair.Value > maxCount)
            {
                maxCount = pair.Value;
                maxChar = pair.Key;
            }
        }

        Console.WriteLine($"Maximum occurring character: '{maxChar}' ({maxCount} times)");
    }
}

    {

16. Implement a C# console application that checks whether the entered email is in proper format or not using regular expressions.

using System;
using System.Text.RegularExpressions;

namespace EmailValidatorApp
{
    class Program
    {
        static void Main()
        {
            Console.Write("Enter your email: ");
            string email = Console.ReadLine();

            string pattern = @"^[^@\s]+@[^@\s]+\.[^@\s]+$";

            if (Regex.IsMatch(email, pattern))
            {
                Console.WriteLine(" Valid email address!");
            }
            else
            {
                Console.WriteLine(" Invalid email address!");
            }

            Console.ReadLine();
        }
    }
}


17. Demonstrate the interface inheritance using explicit interface Implementation.

using System;

namespace InterfaceInheritanceDemo
{
    // Base interface
    interface animal
    {
        void Speak();
    }

    // Derived interface inherits from animal
    interface mammal : animal
    {
        void Walk();
    }

    // Class implementing both interfaces explicitly
    class Dog : mammal
    {
        // Explicit interface implementation
        void animal.Speak()
        {
            Console.WriteLine("The dog barks.");
        }

        void mammal.Walk()
        {
            Console.WriteLine("The dog walks on four legs.");
        }

        // Regular method in the class
        public void Eat()
        {
            Console.WriteLine("The dog eats food.");
        }
    }

    class Program
    {
        static void Main()
        {
            Dog d = new Dog();
            d.Eat(); // normal method

            // Explicit interface methods can only be called using interface references
            animal a = d;
            a.Speak();

            mammal m = d;
            m.Walk();

            Console.ReadLine();
        }
    }
}


18. write a c# console application  to demonstrate  the  delegates concept

using System;

namespace DelegateDemo
{
    // Declare a delegate
    delegate void SimpleDelegate(string message);

    class Program
    {
        // Method that matches the delegate signature
        static void Greet(string msg)
        {
            Console.WriteLine(msg);
        }

        static void Main()
        {
            // Create delegate instance pointing to Greet
            SimpleDelegate del = Greet;

            // Call the method using the delegate
            del("Hello! This is a delegate demonstration.");

            Console.ReadLine();
        }
    }
}
