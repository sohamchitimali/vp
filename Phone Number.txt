using System;
using System.Text.RegularExpressions;

public partial class EmailPhone : System.Web.UI.Page
{
    protected void btnValidate_Click(object sender, EventArgs e)
    {
        string email = txtEmail.Text;
        string phone = txtPhone.Text;

        // Regex patterns
        string emailPattern = @"^[^@\s]+@[^@\s]+\.[^@\s]+$";
        string phonePattern = @"^[6-9]\d{9}$";  // Indian 10-digit number

        bool isEmailValid = Regex.IsMatch(email, emailPattern);
        bool isPhoneValid = Regex.IsMatch(phone, phonePattern);

        if (isEmailValid && isPhoneValid)
            lblResult.Text = "Both Email and Phone Number are valid!";
        else if (!isEmailValid)
            lblResult.Text = "Invalid Email Address!";
        else
            lblResult.Text = "Invalid Phone Number!";
    }
}
